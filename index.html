Encontrar un número entre 1 - 1000

código:


import java.util.Scanner;


public class adivina {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int min = 0;
        int max = 1000;
        int n;
        String r, m;
        System.out.println("Piensa en un número y lo adivinaré");
        do {
            n = (int)(Math.random()*(max-min) + min);
            System.out.println("¿Tú número es "+n+"?");
            System.out.print("Ingresa s/n: ");
            r = sc.next();
            if("s".equals(r)) break;
            System.out.println("¿Tú número es mayor o menor?");
            System.out.print("Ingresa mayor / menor: ");
            m = sc.next();
            System.out.println("");
            if ("mayor".equals(m)) min = n+1;
            if ("menor".equals(m)) max = n-1;
        } while ("n".equals(r));
        System.out.println("Tu numero es: "+n);
    }
}
